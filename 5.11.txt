SELECT MIN(PLACA), MAX(PLACA), AVG(PLACA), COUNT(*) 
FROM RADNIK WHERE SODJEL=10;

SELECT SUM(PLACA), SUM(DODATAK) 
FROM RADNIK WHERE POSAO='TRG. PUTNIK';

SELECT MIN(PLACA), MAX(PLACA), AVG(PLACA), COUNT(*), SODJEL 
FROM RADNIK GROUP BY SODJEL;
//GRUPIRANJE PO ODJELU

SELECT SODJEL, POSAO, COUNT(*) FROM RADNIK 
GROUP BY SODJEL, POSAO;

SELECT SODJEL, POSAO, COUNT(*) FROM RADNIK
GROUP BY SODJEL, POSAO
HAVING COUNT(*)>2;


SELECT AVG(PLACA+DODATAK)*12 FROM RADNIK 
WHERE POSAO='TRG. PUTNIK';

SELECT AVG(PLACA)*12 FROM RADNIK
WHERE POSAO NOT IN ('DIREKTOR', 'VODITELJ')
GROUP BY SODJEL;

SELECT IME, PREZIME, 
ROUND (PLACA/(22*8),2) "ZARADA PO SATU" FROM RADNIK
WHERE PLACA/(22*8)>100;

SELECT IME, PREZIME, PLACA, DODATAK, PLACA+NVL(DODATAK,0) 
FROM RADNIK WHERE SODJEL=30;

SELECT AVG(PLACA), AVG(DODATAK), AVG(PLACA+DODATAK), AVG(PLACA+NVL(DODATAK,0)) 
FROM RADNIK WHERE SODJEL=30;

SELECT IME || ', ' || POSAO RADNIK FROM RADNIK
WHERE SODJEL=30 
ORDER BY POSAO;

SELECT PREZIME, LENGTH(PREZIME) "DUZINA" FROM RADNIK;

SELECT IME, POSAO, DECODE(POSAO, 'ANALITICAR', 1, 'VODITELJ', 3, 
'DIREKTOR', 5, 2) "KLASA" FROM RADNIK;




SELECT MAX(LENGTH(IME)) DUZINA FROM RADNIK;

SELECT AVG(PLACA), MIN(PLACA), MAX(PLACA) FROM RADNIK WHERE
POSAO='ANALITICAR' OR POSAO='VODITELJ';

SELECT IME, PLACA, PLACA/22 DNEVNA, PLACA/22/8 PO_SATU 
FROM RADNIK WHERE DODATAK IS NULL;

SELECT SODJEL, SUM(PLACA/22) FROM RADNIK WHERE POSAO='ANALITICAR' 
GROUP BY SODJEL;

SELECT POSAO, SODJEL, DECODE(SODJEL, 10, 'KADROVI', 20, 'ANALIZA', 30, 'DIZAJN', 'RAZNO') 
OPIS FROM RADNIK;

SELECT PLACA, SODJEL, COUNT(*) FROM RADNIK GROUP BY SODJEL, PLACA HAVING COUNT(*)>=2;

SELECT SODJEL, (MAX(PLACA) - MIN(PLACA)) "RAZLIKA" FROM RADNIK
GROUP BY SODJEL;

SELECT POSAO, AVG(PLACA+NVL(DODATAK,0)) ZARADA FROM RADNIK GROUP BY POSAO;

SELECT AVG(PLACA) FROM RADNIK WHERE DODATAK IS NOT NULL;

SELECT COUNT(PLACA), COUNT(DODATAK) FROM RADNIK WHERE SODJEL=20;
