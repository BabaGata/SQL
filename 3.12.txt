zadaci:
1. SELECT IME, LOKACIJA FROM RADNIK, ODJEL WHERE RADNIK.SODJEL=ODJEL.SODJEL
ORDER BY LOKACIJA ASC;

2. SELECT POD.IME, POD.PLACA, NAD.IME, NAD.PLACA FROM RADNIK POD, RADNIK NAD
WHERE POD.SNADR=NAD.SRADNIK AND POD.PLACA>NAD.PLACA;

3. SELECT AVG(POD.PLACA) PROSJECNA_PLACA, COUNT(POD.SRADNIK) BR_RADNIKA FROM RADNIK 
POD, RADNIK NAD WHERE POD.PLACA>NAD.PLACA AND POD.SNADR=NAD.SRADNIK;

4. SELECT POD.IME, POD.PLACA, POD.DATZAP, NAD.SRADNIK SEF_RADNIK, 
NAD.IME SEF_IME, NAD.PLACA SEF_PLACA, NAD.DATZAP SEF_DATZAP FROM RADNIK POD, RADNIK NAD 
WHERE POD.SNADR=NAD.SRADNIK AND POD.DATZAP<NAD.DATZAP;

5. SELECT NAZIV, LOKACIJA, IME, PLACA, POSAO FROM ODJEL, RADNIK
WHERE RADNIK.SODJEL(+)=ODJEL.SODJEL;
