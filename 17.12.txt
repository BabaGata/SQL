AVG (DECODE(LOKACIJA, 'RIJEKA', PLACA, 0)) "RIJEKA",
AVG (DECODE(LOKACIJA, 'MATULJI', PLACA, 0)) "MATULJI",
AVG (DECODE(LOKACIJA, 'VISKOVO', PLACA, 0)) "VISKOVO",
AVG (PLACA) "PROSJEK ZA POSAO",
AVG (0) POMOCNI
FROM RADNIK, ODJEL WHERE RADNIK.SODJEL=ODJEL.SODJEL
GROUP BY POSAO
ORDER BY POSAO;
